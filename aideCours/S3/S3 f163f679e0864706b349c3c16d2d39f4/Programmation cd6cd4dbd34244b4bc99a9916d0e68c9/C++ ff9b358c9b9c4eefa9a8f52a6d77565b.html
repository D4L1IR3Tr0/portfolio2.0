/*
 Lorsque vous passez un tableau à une fonction en C++, même si vous n'utilisez pas une référence explicite,
 en réalité, le tableau est toujours passé par référence. Cela signifie que toute modification apportée au
 tableau à l'intérieur de la fonction se répercute sur le tableau d'origine dans la fonction appelante.
 */


void tri(date t[], int n) {
    int indmin;
    date min(0, 0, 0);
    for (int i = 0; i < n; i++) {
        indmin = i;
        for (int j = i + 1; j < n; j++) {
            if (t[j] < t[indmin]) {
                indmin = j;
            }
        }
        min = t[indmin];
        t[indmin] = t[i];
        t[i] = min;
    }
}

int main() {
    int n = 2;
    date tabDate[] = {{14, 04, 1998}, {12, 02, 1995}};


    std::cout << "Avant tri : "<< std::endl;
    for (int i = 0; i < n; i++) {
        tabDate[i].affiche();
        cout << endl;

    tri(tabDate,n);

    std::cout << "apres tri : "<< std::endl;
    for (int i = 0; i < n; i++) {

#include <iostream>
#include <vector>

// Fonction pour fusionner deux sous-listes triées
void merge(std::vector<int>& arr, int left, int mid, int right) {
    // Calculer la taille des deux sous-listes
    int n1 = mid - left + 1;
    int n2 = right - mid;

    // Créer des sous-listes temporaires
    std::vector<int> leftArr(n1);
    std::vector<int> rightArr(n2);

    // Copier les éléments dans les sous-listes temporaires
    for (int i = 0; i < n1; i++)
        leftArr[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        rightArr[j] = arr[mid + 1 + j];

    // Fusionner les sous-listes en ordre
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (leftArr[i] <= rightArr[j]) {
            arr[k] = leftArr[i];
            i++;
        } else {
            arr[k] = rightArr[j];
            j++;
        }
        k++;
    }

    // Copier les éléments restants, s'il y en a
    while (i < n1) {
        arr[k] = leftArr[i];
        i++;
        k++;
    }
    while (j < n2) {
        arr[k] = rightArr[j];
        j++;
        k++;
    }
}

// Fonction de tri fusion
void mergeSort(std::vector<int>& arr, int left, int right) {
    if (left < right) {
        // Trouver le point médian du tableau
        int mid = left + (right - left) / 2;

        // Trier récursivement les deux moitiés
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);

        // Fusionner les moitiés triées
        merge(arr, left, mid, right);
    }
}

int main() {
    std::vector<int> arr = {12, 11, 13, 5, 6, 7};
    int arrSize = arr.size();

    std::cout << "Tableau non trié : ";
    for (int i = 0; i < arrSize; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;

    // Appeler la fonction mergeSort pour trier le tableau
    mergeSort(arr, 0, arrSize - 1);

    std::cout << "Tableau trié : ";
    for (int i = 0; i < arrSize; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;

    return 0;
}

#include <iostream>
#include <vector>

void triInsertion(std::vector<int>& tableau) {
    int n = tableau.size();

    for (int i = 1; i < n; i++) {
        int cle = tableau[i];
        int j = i - 1;

        // Déplace les éléments du tableau qui sont plus grands que la clé
        // vers la droite d'une position.
        while (j >= 0 && tableau[j] > cle) {
            tableau[j + 1] = tableau[j];
            j--;
        }

        tableau[j + 1] = cle;
    }
}

int main() {
    std::vector<int> tableau = {12, 11, 13, 5, 6};

    std::cout << "Tableau non trié : ";
    for (int i : tableau) {
        std::cout << i << " ";
    }
    std::cout << std::endl;

    triInsertion(tableau);

    std::cout << "Tableau trié : ";
    for (int i : tableau) {
        std::cout << i << " ";
    }
    std::cout << std::endl;

    return 0;
}

#include <iostream>

using namespace std;

class Cellule {
public:
    int info;
    Cellule* suivant;

    Cellule(int info, Cellule* suivant = NULL) {
        this->info = info;
        this->suivant = suivant;
    }
};

class Liste {
private:
    Cellule* ad;

public:
    Liste() : ad(nullptr) {}


    void ajouter(int info) {
        Cellule* nouvelleCellule = new Cellule(info, ad);
        ad = nouvelleCellule;
    }


    void afficher() {
        Cellule* compteur = ad;
        while (compteur != nullptr) {
            cout << compteur->info << " ";
            compteur = compteur->suivant;
        }
        cout << endl;
    }


    void vider() {
        while (ad != nullptr) {
            Cellule* temp = ad;
            ad = ad->suivant;
            delete temp;
        }
    }

    
    ~Liste() {
        vider();
    }
};

int main() {
    Liste maListe;

    maListe.ajouter(3);
    maListe.ajouter(2);
    maListe.ajouter(1);

    maListe.afficher();

    // N'oubliez pas de libérer la mémoire occupée par la liste
    maListe.vider();

    return 0;
}
