void tri(int t[], int n) {
    int indmin;
    int min;
    for (int i = 0; i < n; i++) {
        indmin = i;
        for (int j = i + 1; j < n; j++) {
            if (t[j] < t[indmin]) {
                indmin = j;
            }
        }
        min = t[indmin];
        t[indmin] = t[i];
        t[i] = min;
    }
}

class TableauDynamique {
public:
    // Constructeur
    TableauDynamique() {}

    // Méthode pour vider le tableau
    void vider() {
        tableau.clear();
    }

    // Méthode pour ajouter un élément au tableau
    void ajouterElement(int element) {
        tableau.push_back(element);
    }

    // Méthode pour accéder à un élément du tableau
    int accesElement(int i) {
        if (i >= 0 && i < tableau.size()) {
            return tableau[i];
        } else {
            return -1; // Vous pouvez également lever une exception ici si vous le souhaitez
        }
    }

    // Méthode pour modifier un élément du tableau
    void modifierElement(int i, int nouvelElement) {
        if (i >= 0 && i < tableau.size()) {
            tableau[i] = nouvelElement;
        } else {
            std::cout << "Index hors limites" << std::endl;
        }
    }

    // Méthode pour afficher le tableau
    void afficher() {
        for (int element : tableau) {
            std::cout << element << " ";
        }
        std::cout << std::endl;
    }

    // Méthode pour supprimer un élément du tableau
    void supprimerElement(int element) {
        auto it = std::find(tableau.begin(), tableau.end(), element);
        if (it != tableau.end()) {
            tableau.erase(it);
        } else {
            std::cout << "L'élément n'est pas dans le tableau" << std::endl;
        }
    }

    // Méthode pour agrandir le tableau en ajoutant une certaine quantité d'éléments
    void agrandir(int taille) {
        tableau.resize(tableau.size() + taille);
    }

private:
    std::vector<int> tableau;
};

Manipulation et modification des Listes Doublement Chainées

class Node {
public:
    int data;
    Node* next;
    Node* prev;

    Node(int value) : data(value), next(nullptr), prev(nullptr) {}
};

class DoubleLinkedList {
public:
    // Constructeur
    DoubleLinkedList() : head(nullptr), tail(nullptr), size(0) {}

    // Destructeur
    ~DoubleLinkedList() {
        vider();
    }

    // Affichage de la liste de gauche à droite
    void afficherDebut() const {
        Node* current = head;
        while (current != nullptr) {
            std::cout << current->data << " ";
            current = current->next;
        }
        std::cout << std::endl;
    }

    // Affichage de la liste de droite à gauche
    void afficherFin() const {
        Node* current = tail;
        while (current != nullptr) {
            std::cout << current->data << " ";
            current = current->prev;
        }
        std::cout << std::endl;
    }

    // Test si la liste est vide
    bool estVide() const {
        return size == 0;
    }

    // Ajout d'un élément en tête de liste
    void ajouterEnTete(int element) {
        Node* newNode = new Node(element);
        if (estVide()) {
            head = tail = newNode;
        } else {
            newNode->next = head;
            head->prev = newNode;
            head = newNode;
        }
        size++;
    }

    // Ajout d'un élément en queue de liste
    void ajouterEnQueue(int element) {
        Node* newNode = new Node(element);
        if (estVide()) {
            head = tail = newNode;
        } else {
            newNode->prev = tail;
            tail->next = newNode;
            tail = newNode;
        }
        size++;
    }

    // Suppression de l'élément de tête
    void supprimerEnTete() {
        if (!estVide()) {
            Node* temp = head;
            head = head->next;
            if (head != nullptr) {
                head->prev = nullptr;
            } else {
                tail = nullptr;
            }
            delete temp;
            size--;
        }
    }

    // Vider la liste
    void vider() {
        while (!estVide()) {
            supprimerEnTete();
        }
    }

    // Renvoie le nombre d'éléments dans la liste
    int nombreElements() const {
        return size;
    }

    // Accès au i-ème élément
    int& accesElement(int i) const {
        Node* current = head;
        int index = 0;
        while (current != nullptr && index < i) {
            current = current->next;
            index++;
        }
        if (current != nullptr) {
            return current->data;
        } else {
            throw std::out_of_range("Index hors limites");
        }
    }

    // Modification du i-ème élément
    void modifierElement(int i, int nouvelElement) {
        Node* current = head;
        int index = 0;
        while (current != nullptr && index < i) {
            current = current->next;
            index++;
        }
        if (current != nullptr) {
            current->data = nouvelElement;
        } else {
            throw std::out_of_range("Index hors limites");
        }
    }

private:
    Node* head;
    Node* tail;
    int size;
};

Manipulation et modification des Piles

class Pile {
public:
    // Empiler un élément sur la pile
    void empiler(void* element) {
        pile.push(element);
    }

    // Récupérer le sommet de la pile
    void* sommetDeLaPile() {
        if (!pile.empty()) {
            return pile.top();
        } else {
            return nullptr;
        }
    }

    // Dépiler un élément de la pile
    void* depiler() {
        if (!pile.empty()) {
            void* sommet = pile.top();
            pile.pop();
            return sommet;
        } else {
            return nullptr;
        }
    }

    // Vérifier si la pile est vide
    bool estVide() const {
        return pile.empty();
    }

    // Vider la pile
    void vider() {
        while (!pile.empty()) {
            pile.pop();
        }
    }

    // Destructeur
    ~Pile() {
        vider();
    }

private:
    std::stack<void*> pile;
};

Manipulation et modification des Files

class File {
public:
    // Enfiler un élément dans la file
    void enfiler(void* element) {
        file.push(element);
    }

    // Récupérer le premier élément de la file
    void* premierDeLaFile() {
        if (!file.empty()) {
            void* premier = file.front();
            file.pop();
            return premier;
        } else {
            return nullptr;
        }
    }

    // Vérifier si la file est vide
    bool estVide() const {
        return file.empty();
    }

    // Vider la file
    void vider() {
        while (!file.empty()) {
            file.pop();
        }
    }

    // Destructeur
    ~File() {
        vider();
    }

private:
    std::queue<void*> file;
};

Manipulation et modification des Arbres Binaires

class Noeud {
public:
    int valeur;
    Noeud* gauche;
    Noeud* droite;

    Noeud(int valeur) : valeur(valeur), gauche(nullptr), droite(nullptr) {}
};

class ArbreBinaire {
public:
    // Constructeur
    ArbreBinaire() : racine(nullptr) {}

    // Destructeur
    ~ArbreBinaire() {
        vider(racine);
    }

    // Vider l'arbre
    void vider(Noeud* noeud) {
        if (noeud != nullptr) {
            vider(noeud->gauche);
            vider(noeud->droite);
            delete noeud;
        }
    }

    // Vérifier si l'arbre est vide
    bool estVide() const {
        return racine == nullptr;
    }

    // Insérer un élément dans l'arbre
    void insererElement(int valeur) {
        racine = insererElementRecursive(racine, valeur);
    }

    // Fonction récursive pour insérer un élément
    Noeud* insererElementRecursive(Noeud* noeud, int valeur) {
        if (noeud == nullptr) {
            return new Noeud(valeur);
        }

        if (valeur < noeud->valeur) {
            noeud->gauche = insererElementRecursive(noeud->gauche, valeur);
        } else if (valeur > noeud->valeur) {
            noeud->droite = insererElementRecursive(noeud->droite, valeur);
        }

        return noeud;
    }

    // Afficher l'arbre par parcours infixe
    void afficherParcoursInfixe() const {
        afficherParcoursInfixeRecursive(racine);
        std::cout << std::endl;
    }

    // Fonction récursive pour afficher par parcours infixe
    void afficherParcoursInfixeRecursive(const Noeud* noeud) const {
        if (noeud != nullptr) {
            afficherParcoursInfixeRecursive(noeud->gauche);
            std::cout << noeud->valeur << " ";
            afficherParcoursInfixeRecursive(noeud->droite);
        }
    }

    // Rechercher un élément dans l'arbre
    bool rechercherElement(int valeur) const {
        return rechercherElementRecursive(racine, valeur);
    }

    // Fonction récursive pour rechercher un élément
    bool rechercherElementRecursive(const Noeud* noeud, int valeur) const {
        if (noeud == nullptr) {
            return false;
        }

        if (valeur == noeud->valeur) {
            return true;
        } else if (valeur < noeud->valeur) {
            return rechercherElementRecursive(noeud->gauche, valeur);
        } else {
            return rechercherElementRecursive(noeud->droite, valeur);
        }
    }

    // Calculer la hauteur de l'arbre
    int hauteurArbre() const {
        return hauteurArbreRecursive(racine);
    }

    // Fonction récursive pour calculer la hauteur de l'arbre
    int hauteurArbreRecursive(const Noeud* noeud) const {
        if (noeud == nullptr) {
            return 0;
        }

        int hauteurGauche = hauteurArbreRecursive(noeud->gauche);
        int hauteurDroite = hauteurArbreRecursive(noeud->droite);

        return std::max(hauteurGauche, hauteurDroite) + 1;
    }

private:
    Noeud* racine;
};
